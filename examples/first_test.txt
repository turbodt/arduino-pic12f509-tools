; Example command
;
; This program blinks a led
;

setup
  movwf osccal
  call set_up_wait
  movlw 0xFD
  tris 0x06
loop
  movlw 0x02
  movwf gpio   ; gpio is a default location
  call wait
  movlw 0x00
  movwf gpio
  call wait
  goto loop
end
  sleep



set_up_wait
  clr f 07h
  clr f 08h
  retlw 0x00

; value on 07h determines how many times the loop is done
wait
  decfsz f 08h
  nop
  goto wait
  decfsz f 07h
  goto wait
  retlw 0x00
